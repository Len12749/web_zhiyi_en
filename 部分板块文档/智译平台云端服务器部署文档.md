# 智译平台云端服务器部署文档



## 配置文件

```cmd
server {
    listen 80;
    listen [::]:80;

    # 解决大文件上传问题
    client_max_body_size 800M;

    # 基础超时设置
    proxy_connect_timeout 60s;
    proxy_read_timeout 3600s;
    proxy_send_timeout 3600s;

    # 如需绑定域名，请取消注释并修改
    # server_name your-domain.com www.your-domain.com;
    # 或者使用服务器公网IP
    # server_name 1.2.3.4;  # 替换为你的服务器公网IP

    # SSE 专用配置 - 关键修复
    location /api/sse/ {
        proxy_pass http://100.85.18.58:3000;

        # SSE 关键配置
        proxy_buffering off;
        proxy_cache off;
        proxy_set_header Connection '';
        proxy_http_version 1.1;

        # SSE 专用超时设置（更长的超时时间）
        proxy_read_timeout 86400s;  # 24小时
        proxy_send_timeout 86400s;  # 24小时

        # 头部设置
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # 禁用所有缓存
        add_header Cache-Control "no-cache, no-store, must-revalidate";
        add_header Pragma "no-cache";
        add_header Expires "0";

        # 确保 SSE 流不被缓冲
        proxy_set_header X-Accel-Buffering no;
    }

    # API 路由配置（非 SSE）
    location /api/ {
        proxy_pass http://100.85.18.58:3000;

        # 代理头信息
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # HTTP/1.1 支持
        proxy_http_version 1.1;
        proxy_set_header Connection "";

        # 不缓存 API 响应
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate";
    }

    # 静态文件配置（启用缓存）
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot)$ {
        proxy_pass http://100.85.18.58:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;

        # 静态资源缓存设置（7天）
        expires 7d;
        add_header Cache-Control "public, max-age=604800";
        access_log on;

        # 启用代理缓存
        proxy_cache static_cache;
        proxy_cache_valid 200 7d;
        proxy_cache_bypass $http_pragma;
        proxy_cache_revalidate on;
    }

    # 处理其他页面请求
    location / {
        proxy_pass http://100.85.18.58:3000;

        # 代理头信息
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # HTTP/1.1 支持（不设置 WebSocket 的 upgrade）
        proxy_http_version 1.1;
        proxy_set_header Connection "";

        # 不缓存动态内容
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate";
    }
}

# 定义静态缓存区（需放在server块外部）
proxy_cache_path /var/cache/nginx/static levels=1:2 keys_zone=static_cache:10m max_size=10g inactive=30d use_temp_path=off;
```
